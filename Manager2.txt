// The Manager2 class implements the following requirements:
// 1/ The system should classify the patient's respiratory status based on oxygen saturation (O2) and respiration rate (RR) readings as low O2 (O2 < 88), fast breathing (RR > 15), or normal. 
// 2/ The system should separately classify the patient's breathing frequency based on RR reading as normal (12-20 breaths per minute), suspicious (< 6 br/min), slow (6-12 br/min), or fast (> 20 br/min).
// 3/ The system should output both results for further processing.


import java.io.*;
import java.io.IOException;
import java.util.*;

public class Manager2 {

    private static List<String> LOGS = new ArrayList<>();

    private static final double MIN_WEIGHT = 50;
    private static final double MAX_WEIGHT = 120;

    private static boolean debug = false;

    private String last_stat = null;


    public void method1(Patient pt, int rr, double o2) {

        if (pt.w < MIN_WEIGHT || pt.w > MAX_WEIGHT) {
            System.out.println("Suspicious weight!");
            LOGS.add("Invalid weight for " + pt.patient_name);
        }

        if (o2 < 88) {
            last_stat = "LOW O2";
        } else if (rr > 15) {
            last_stat = "FAST BREATHING";
        } else {
            last_stat = "OK";
        }

        LOGS.add("Processed " + pt.patient_name);
    }


    public int method2(Patient p, int rr, boolean print) {

        int cls = -999;

        if (p.w <= 0 || p.w > 1000) {
            LOGS.add("Bad weight: " + p.w);
            return cls;
        }

        if (rr > 20) {
            cls = 3;
        } else if (rr > 12) {
            cls = 1;
        } else if (rr < 6) {
            cls = 2;
        } else {
            cls = 0;
        }

        if (print) {
            System.out.println("Class result=" + cls + " for " + p.patient_name);
        }

        return cls;
    }

}


class Patient {
    public String patient_name;
    public double w;
    public String cond;

    public Patient(String name, double weight, String condition) {
        this.patient_name = name;
        this.w = weight;
        this.cond = condition;
    }
}
